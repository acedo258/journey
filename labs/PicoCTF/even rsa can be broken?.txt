from Crypto.Util.number import long_to_bytes, inverse
import sympy

N = 23884992828262274549190305705247306140924527877595855715615081073100623967908892044636395843252370233301794797889950357560068475818738315450155346106665498

# 1. Factorizamos N
factors = sympy.factorint(N)
p, q = list(factors.keys())

# 2. Calculamos phi(N)
phi = (p-1)*(q-1)

# 3. Calculamos d
d = inverse(e, phi)

# 4. Desciframos
m = pow(c, d, N)
flag = long_to_bytes(m)
print(flag)
